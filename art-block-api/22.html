function spr(e) {

     return c.map(e => parseInt(e, 16))

function gS(e) {

function rnd() {
     return seed ^= seed << 13, seed ^= seed >> 17, ((seed ^= seed << 5) < 0 ? 1 + ~seed : seed) % 1e3 / 1e3
}

function pk(e, c) {
     return c[Math.max(0, Math.floor(e / 255 * c.length - 1e-6))]
}

function rng(e, c) {
     return rnd() * (c - e) + e
}

function rngFlr(e, c) {
     return Math.floor(rng(e, c))
}

function pick(e) {
     return e[rngFlr(0, e.length)]

function map(e, c, a, r, n) {
     return (e - c) / (a - c) * (n - r) + r
