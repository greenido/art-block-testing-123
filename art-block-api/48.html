          this.word = a.toLowerCase(), this.start_y = b, this.height = c, this.edge_padding = d, this.padding = gs(1 / (3 * this.word.length)), this.letter_width = (width - 2 * this.edge_padding) / this.word.length - this.padding, this.letters = [], this.prepare()
     }
     prepare() {
          for (let a = 0; a < this.word.length; a++) this.letters.push(new Letter(this.word[a], a * this.letter_width + (this.edge_padding + this.padding / 2) + this.padding * a, this.start_y, this.letter_width, this.height))
     }
     display() {
          colorMode(HSB, 360, 100, 100, 1), noFill();
          for (let a = 0; 3 > a; a++) {
               0 == a ? strokeWeight(gs(.22 / this.word.length)) : 1 === a ? strokeWeight(gs(.18 / this.word.length)) : 2 === a ? strokeWeight(gs(.09 / this.word.length)) : void 0;
               for (let b = 0; b < this.letters.length; b++) 0 == a % 2 ? stroke(this.letters[b].get_color2()) : stroke(this.letters[b].get_color()), this.letters[b].display()
          }
     }
}
class Letter {
     constructor(a, b, c, d, e) {
          this.letter = a, this.x = b, this.y = c, this.w = d, this.h = e, colorMode(HSB, 360, 100, 100);
          let f = random(360),
               g = (f + 128) % 360;
          switch (floor(map(hp[4], 0, 255, 0, 3.999))) {
               case 0:
                    this.col = color(f, 80, 0), this.col2 = color(g, 80, 100);
                    break;
               case 1:
                    this.col = color(f, 80, 100), this.col2 = color(g, 80, 90);
                    break;
               case 2:
                    this.col = color(f, 0, 100), this.col2 = color(g, 0, 0);
                    break;
               case 3:
                    this.col = color(f, 0, 0), this.col2 = color(g, 0, 100);
          }
     }
     get_color() {
          return this.col
     }
     get_color2() {
          return this.col2
     }
     display() {
          push(), translate(this.x, this.y), this.fetch_letter(), pop()
     }
     update_location(a, b) {
          this.x = a, this.y = b
     }
     fetch_letter() {
          let a = this.w,
               b = this.h;
          switch (this.letter) {
               case "a":
                    line(0, b, a / 2, 0), line(a / 2, 0, a, b), line(a / 2 / 2, b / 2, (a + a / 2) / 2, b / 2);
                    break;
               case "b":
                    line(0, 0, 0, b), arc(a / 2, .25 * b, a, b / 2, PI + HALF_PI, HALF_PI), arc(a / 2, .75 * b, a, b / 2, PI + HALF_PI, HALF_PI), line(0, 0, a / 2, 0), line(0, b / 2, a / 2, b / 2), line(0, b, a / 2, b);
                    break;
               case "c":
                    arc(a / 2, b / 2, a, b, QUARTER_PI + QUARTER_PI / 2, TWO_PI - QUARTER_PI - QUARTER_PI / 2);
                    break;
               case "d":
                    line(0, 0, 0, b), arc(a / 2, b / 2, a, b, PI + HALF_PI, HALF_PI), line(0, 0, a / 2, 0), line(0, b, a / 2, b);
                    break;
               case "e":
                    line(0, 0, 0, b), line(0, 0, a, 0), line(0, b / 2, a, b / 2), line(0, b, a, b);
                    break;
               case "f":
                    line(0, 0, 0, b), line(0, 0, a, 0), line(0, b / 2, a, b / 2);
                    break;
               case "g":
                    arc(a / 2, b / 2, a, b, 0, TWO_PI - QUARTER_PI), line(a / 2, b / 2, a, b / 2);
                    break;
               case "h":
                    line(0, 0, 0, b), line(a, 0, a, b), line(0, b / 2, a, b / 2);
                    break;
               case "i":
                    line(0, 0, a, 0), line(a / 2, 0, a / 2, b), line(0, b, a, b);
                    break;
               case "j":
                    line(0, 0, a, 0), line(a - a / 4, 0, a - a / 4, b / 2), arc(a / 4, b / 2, a, b, 0, HALF_PI + QUARTER_PI / 3);
                    break;
               case "k":
                    line(a, 0, 0, b / 2), line(a, b, 0, b / 2), line(0, 0, 0, b);
                    break;
               case "l":
                    line(0, 0, 0, b), line(0, b, a, b);
                    break;
               case "m":
                    line(0, b, a / 4, 0), line(a, b, a - a / 4, 0), line(a / 2, b - b / 4, a / 4, 0), line(a / 2, b - b / 4, a - a / 4, 0);
                    break;
               case "n":
                    line(0, 0, 0, b), line(0, 0, a, b), line(a, 0, a, b);
                    break;
               case "o":
                    ellipse(a / 2, b / 2, a, b);
                    break;
               case "p":
                    line(0, 0, 0, b), arc(a / 2, b / 4, a, b / 2, PI + HALF_PI, HALF_PI), line(0, 0, a / 2, 0), line(0, b / 2, a / 2, b / 2);
                    break;
               case "q":
                    ellipse(a / 2, b / 2, a, b), line(a / 2, b / 2, a, b);
                    break;
               case "r":
                    line(0, 0, 0, b), arc(a / 2, b / 4, a, b / 2, PI + HALF_PI, HALF_PI), line(0, 0, a / 2, 0), line(0, b / 2, a / 2, b / 2), line(a / 2, b / 2, a, b);
                    break;
               case "s":
                    arc(a / 2, b / 4, a, b / 2, HALF_PI, TWO_PI - QUARTER_PI), arc(a / 2, b - b / 4, a, b / 2, PI + HALF_PI, PI);
                    break;
               case "t":
                    line(0, 0, a, 0), line(a / 2, 0, a / 2, b);
                    break;
               case "u":
                    arc(a / 2, b - b / 4, a, b / 2, 0, PI), line(0, 0, 0, b - b / 4), line(a, 0, a, b - b / 4);
                    break;
               case "v":
                    line(0, 0, a / 2, b), line(a, 0, a / 2, b);
                    break;
               case "w":
                    line(0, 0, a / 4, b), line(a, 0, a - a / 4, b), line(a / 4, b, a / 2, b / 4), line(a - a / 4, b, a / 2, b / 4);
                    break;
               case "x":
                    line(0, 0, a, b), line(a, 0, 0, b);
                    break;
               case "y":
                    line(0, 0, a / 2, b / 2), line(a, 0, a / 2, b / 2), line(a / 2, b / 2, a / 2, b);
                    break;
               case "z":
                    line(0, 0, a, 0), line(a, 0, 0, b), line(0, b, a, b);
                    break;
               case "0":
                    ellipse(a / 2, b / 2, a, b), line(a / 2 + sin(-QUARTER_PI) * a / 2, b / 2 + cos(-QUARTER_PI) * b / 2, a / 2 + sin(PI - QUARTER_PI) * a / 2, b / 2 + cos(PI - QUARTER_PI) * b / 2);
                    break;
               case "1":
                    line(a / 2, 0, a / 2, b), line(a / 2, 0, a / 4, b / 4), line(0, b, a, b);
                    break;
               case "2":
                    arc(a / 2, b / 4, a, b / 2, PI, TWO_PI), line(a / 2 + sin(HALF_PI) * a / 2, b / 4 + cos(HALF_PI) * b / 4, 0, b), line(0, b, a, b);
                    break;
               case "3":
                    line(0, 0, a, 0), line(a, 0, a / 2, b / 2), arc(a / 2, b - b / 4, a, b / 2, PI + HALF_PI, PI);
                    break;
               case "4":
                    line(a - a / 4, 0, 0, b / 2), line(a - a / 4, 0, a - a / 4, b), line(0, b / 2, a, b / 2);
                    break;
               case "5":
                    line(0, 0, a, 0), line(0, 0, 0, b / 2), line(0, b / 2, a / 2, b / 2), arc(a / 2, b - b / 4, a, b / 2, PI + HALF_PI, PI);
                    break;
               case "6":
                    arc(a / 2, b / 4, a, b / 2, PI, PI + HALF_PI + QUARTER_PI), line(0, b / 4, 0, b - b / 4), ellipse(a / 2, b - b / 4, a, b / 2);
                    break;
               case "7":
                    line(0, 0, a, 0), line(a / 2, b, a, 0), line(a / 4, b / 2, a - a / 4, b / 2);
                    break;
               case "8":
                    ellipse(a / 2, b / 4, a, b / 2), ellipse(a / 2, b - b / 4, a, b / 2);
                    break;
               case "9":
                    ellipse(a / 2, b / 4, a, b / 2), line(a, b / 4, a, b);
                    break;
               case "-":
                    line(0, b / 2, a, b / 2);
                    break;
               case ".":
                    point(a / 2, b);
                    break;
               case "_":
                    line(0, b, a, b);
                    break;
               case "*":
                    for (let c = 0; 360 > c; c += 120) line(sin(radians(c)) * a / 4 + a / 2, cos(radians(c)) * b / 4 + b / 3, sin(radians(c + 180)) * a / 4 + a / 2, cos(radians(c + 180)) * b / 4 + b / 3), line(sin(radians(c)) * a / 4 + a / 2, cos(radians(c)) * b / 4 + b / 3, sin(radians(c + 180)) * a / 4 + a / 2, cos(radians(c + 180)) * b / 4 + b / 3), line(sin(radians(c)) * a / 4 + a / 2, cos(radians(c)) * b / 4 + b / 3, sin(radians(c + 180)) * a / 4 + a / 2, cos(radians(c + 180)) * b / 4 + b / 3);
                    break;
               case " ":
                    break;
               default:
          }
     }
}
